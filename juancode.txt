enum AppMode { normal, emergency, pwd }

// ─── Data Models ──────────────────────────────────────────────────────────────

class RoomData {
  final String name;
  final RoomType type;
  final double widthFactor;
  final String? badge;
  const RoomData({
    required this.name,
    required this.type,
    this.widthFactor = 1.0,
    this.badge,
  });
}

class FloorData {
  final int number;
  final String label;
  final List<RoomData> rooms;
  final bool stairsLeft;
  final bool stairsRight;
  const FloorData({
    required this.number,
    required this.label,
    required this.rooms,
    this.stairsLeft = true,
    this.stairsRight = true,
  });
}

class PWDReport {
  final String id;
  final int floor;
  final String location;
  final String issueType;
  final String description;
  final String severity;
  final double xNorm;

  const PWDReport({
    required this.id,
    required this.floor,
    required this.location,
    required this.issueType,
    required this.description,
    required this.severity,
    required this.xNorm,
  });

  Map<String, dynamic> toMap() => {
        'floor': floor,
        'location': location,
        'issue_type': issueType,
        'description': description,
        'severity': severity,
        'x_norm': xNorm,
      };

  static PWDReport fromMap(Map<String, dynamic> m) => PWDReport(
        id: m['id'].toString(),
        floor: m['floor'],
        location: m['location'],
        issueType: m['issue_type'],
        description: m['description'],
        severity: m['severity'],
        xNorm: (m['x_norm'] as num).toDouble(),
      );
}

class FloorRoute {
  final double startX;
  final double endX;
  final String note;
  const FloorRoute({
    required this.startX,
    required this.endX,
    required this.note,
  });
}

// ─── Service Layer ─────────────────────────────────────────────────────────────
class ReportService {
  static final ReportService _instance = ReportService._internal();
  factory ReportService() => _instance;
  ReportService._internal() {
    _reports.addAll(kInitialReports);
  }

  final List<PWDReport> _reports = [];

  // TODO (backend): replace with →

  List<PWDReport> getAllReports() => List.unmodifiable(_reports);

  // TODO (backend): replace with →
  //   final res = await supabase.from('reports').select().eq('floor', floor);
  //   return res.map((r) => PWDReport.fromMap(r)).toList();
  List<PWDReport> getFloorReports(int floor) =>
      _reports.where((r) => r.floor == floor).toList();

  // TODO (backend): replace with →
  //   await supabase.from('reports').insert(report.toMap());

}

// ─── Map Screen ───────────────────────────────────────────────────────────────

class MapScreen extends StatefulWidget {
  const MapScreen({super.key});
  @override
  State<MapScreen> createState() => _MapScreenState();
}

class _MapScreenState extends State<MapScreen> {
  int _floorIndex = 0;
  AppMode _mode = AppMode.normal;
  DisabilityMode _disability = DisabilityMode.none;
  PWDReport? _selectedReport;

  // ADDED: tracks which room was tapped, null when no room selected
  RoomData? _selectedRoom;

  final _transformController = TransformationController();
  final ReportService _service = ReportService();

  static const double _mapHeight = 220.0;
  static const double _corridorH = 35.0;
  static const double _stairsW = 22.0;

  FloorData get _floor => kFloors[_floorIndex];
  List<PWDReport> get _floorReports => _service.getFloorReports(_floor.number);

  @override
  void dispose() {
    _transformController.dispose();
    super.dispose();
  }

  // Computes room rects using the same logic as FloorPlanPainter,
  // so tap coordinates match what is drawn on screen.
  List<(RoomData, Rect)> _computeRoomRects(double mapWidth) {
    const roomH = _mapHeight - _corridorH;
    final roomStartX = _floor.stairsLeft ? _stairsW : 4.0;
    final roomEndX = _floor.stairsRight ? mapWidth - _stairsW : mapWidth - 4.0;
    final totalFactor =
        _floor.rooms.fold(0.0, (s, r) => s + r.widthFactor);
    final unitW = (roomEndX - roomStartX) / totalFactor;

    final result = <(RoomData, Rect)>[];
    double x = roomStartX;
    for (final room in _floor.rooms) {
      final rw = unitW * room.widthFactor;
      // Match painter: Rect.fromLTWH(x + 1, 4, rw - 2, roomH - 8)
      result.add((room, Rect.fromLTWH(x + 1, 4, rw - 2, roomH - 8)));
      x += rw;
    }
    return result;
  }

  void _handleTap(Offset localOffset, double mapWidth) {
    final sceneOffset = _transformController.toScene(localOffset);
    const roomH = _mapHeight - _corridorH;

    // 1. Check report markers first (higher priority than room taps)
    for (final PWDReport report in _floorReports) {
      final double mx = report.xNorm * mapWidth;
      final double my = roomH / 2;
      if ((sceneOffset.dx - mx).abs() < 22 &&
          (sceneOffset.dy - my).abs() < 22) {
        setState(() {
          _selectedReport =
              (_selectedReport?.id == report.id) ? null : report;
          _selectedRoom = null; // close room panel if open
        });
        return;
      }
    }

    // 2. Check room rects — ADDED: room tap now shows info popup
    final rects = _computeRoomRects(mapWidth);
    for (final (RoomData room, Rect rect) in rects) {
      if (rect.contains(sceneOffset)) {
        setState(() {
          _selectedRoom = (_selectedRoom == room) ? null : room;
          _selectedReport = null; // close report panel if open
        });
        return;
      }
    }

    // 3. Tapped empty space — close everything
    setState(() {
      _selectedReport = null;
      _selectedRoom = null;
    });
  }

                DropdownButtonFormField<String>(

                                child: Text(loc.$1,
                                    style: const TextStyle(fontSize: 12)),

class FloorPlanPainter extends CustomPainter {
  final FloorData floor;
  final AppMode mode;
  final DisabilityMode disability;
  final List<PWDReport> reports;
  final PWDReport? selected;

  static const double corridorH = 35.0;
  static const double stairsW = 22.0;

  const FloorPlanPainter({
    required this.floor,
    required this.mode,
    required this.disability,
    required this.reports,
    this.selected,
  });

  bool get _hiContrast => disability == DisabilityMode.visual;

  @override
  void paint(Canvas canvas, Size size) {
    final double w = size.width;
    final double h = size.height;
    final double roomH = h - corridorH;

    // Background
    canvas.drawRect(
      Rect.fromLTWH(0, 0, w, h),
      Paint()
        ..color =
            _hiContrast ? Colors.black : const Color(0xFFF2F2F0),
    );

    // Corridor
    canvas.drawRect(
      Rect.fromLTWH(0, roomH, w, corridorH),
      Paint()
        ..color = _hiContrast
            ? const Color(0xFF333300)
            : const Color(0xFFE0DED8),
    );
    canvas.drawLine(
      Offset(0, roomH),
      Offset(w, roomH),
      Paint()
        ..color =
            _hiContrast ? Colors.yellow : const Color(0xFFBBB9B4)
        ..strokeWidth = 0.8,
    );
    _text(canvas, 'CORRIDOR', w / 2, roomH + corridorH / 2,
        fs: 8,
        color: _hiContrast ? Colors.yellow : const Color(0xFF999999),
        maxW: w);

    // Rooms
    final double roomStartX = floor.stairsLeft ? stairsW : 4.0;
    final double roomEndX =
        floor.stairsRight ? w - stairsW : w - 4.0;
    final double totalFactor = floor.rooms
        .fold(0.0, (double s, RoomData r) => s + r.widthFactor);
    final double unitW = (roomEndX - roomStartX) / totalFactor;

    double x = roomStartX;
    for (final RoomData room in floor.rooms) {
      final double rw = unitW * room.widthFactor;
      _drawRoom(canvas, Rect.fromLTWH(x + 1, 4, rw - 2, roomH - 8), room);
      x += rw;
    }

    // Stairs
    if (floor.stairsLeft) {
      _drawStairs(canvas,
          Rect.fromLTWH(2, 4, stairsW - 4, roomH - 8));
    }
    if (floor.stairsRight) {
      _drawStairs(canvas,
          Rect.fromLTWH(w - stairsW + 2, 4, stairsW - 4, roomH - 8));
    }

    // Route overlay
    if (mode == AppMode.pwd) {
      final FloorRoute? route = switch (disability) {
        DisabilityMode.visual => kVisualRoutes[floor.number],
        DisabilityMode.hearing => kHearingRoutes[floor.number],
        _ => kPwdRoutes[floor.number],
      };
      _drawRoute(canvas, w, roomH, route, const Color(0xFF1D9E75));
      // Stair avoidance X marks in corridor under each stairwell
      if (floor.stairsLeft) {
        _drawAvoidX(canvas, stairsW / 2, roomH + corridorH / 2);
      }
      if (floor.stairsRight) {
        _drawAvoidX(canvas, w - stairsW / 2, roomH + corridorH / 2);
      }
    } else if (mode == AppMode.emergency) {
      _drawRoute(
          canvas, w, roomH, kEmergencyRoutes[floor.number], Colors.red);
    }

    // Outer border
    canvas.drawRect(
      Rect.fromLTWH(0, 0, w, h),
      Paint()
        ..color =
            _hiContrast ? Colors.white : const Color(0xFF999999)
        ..strokeWidth = 1.5
        ..style = PaintingStyle.stroke,
    );

    // PWD report markers
    for (final PWDReport report in reports) {
      _drawMarker(
          canvas,
          Offset(report.xNorm * w, (h - corridorH) / 2),
          report,
          selected?.id == report.id);
    }
  }

  void _drawRoom(Canvas canvas, Rect rect, RoomData room) {
    Color fill;
    if (_hiContrast) {
      fill = Colors.black;
    } else {
      fill = switch (room.type) {
        RoomType.classroom => Colors.white,
        RoomType.lab => const Color(0xFFF0F4FD),
        RoomType.office => const Color(0xFFFFF8EE),
        RoomType.toilet => const Color(0xFFECF8F0),
        RoomType.utility => const Color(0xFFF3F3F1),
      };
    }

    canvas.drawRect(rect, Paint()..color = fill);
    canvas.drawRect(
        rect,
        Paint()
          ..color =
              _hiContrast ? Colors.white : Colors.grey.shade400
          ..strokeWidth = 0.8
          ..style = PaintingStyle.stroke);

    _text(canvas, room.name, rect.center.dx, rect.center.dy,
        fs: 8,
        color: _hiContrast ? Colors.white : Colors.black87,
        maxW: rect.width - 4);

    // Badge: dark circle with white letter, matching reference image style
    if (room.badge != null) {
      const double badgeR = 9.0;
      final double bx = rect.right - badgeR - 2;
      final double by = rect.top + badgeR + 3;
      canvas.drawCircle(
          Offset(bx, by),
          badgeR,
          Paint()
            ..color =
                _hiContrast ? Colors.white : const Color(0xFF1A1A2E));
      _text(canvas, room.badge!, bx, by,
          fs: 8,
          color: _hiContrast ? Colors.black : Colors.white,
          bold: true,
          maxW: badgeR * 2);
    }
  }

  void _drawStairs(Canvas canvas, Rect rect) {
    canvas.drawRect(
        rect,
        Paint()
          ..color = _hiContrast
              ? const Color(0xFF222200)
              : const Color(0xFFDDDDDB));
    final Paint stepP = Paint()
      ..color = _hiContrast ? Colors.yellow : Colors.black26
      ..strokeWidth = 0.7;
    for (int i = 1; i < 5; i++) {
      final double y = rect.top + (rect.height / 5) * i;
      canvas.drawLine(
          Offset(rect.left, y), Offset(rect.right, y), stepP);
    }
    _text(canvas, '▲', rect.center.dx, rect.center.dy,
        fs: 10,
        color: _hiContrast ? Colors.yellow : Colors.black38,
        maxW: rect.width);
  }

  void _drawAvoidX(Canvas canvas, double cx, double cy) {
    const double r = 7.0;
    final Paint paint = Paint()
      ..color = const Color(0xFFE24B4A)
      ..strokeWidth = 2
      ..strokeCap = StrokeCap.round;
    canvas.drawLine(
        Offset(cx - r, cy - r), Offset(cx + r, cy + r), paint);
    canvas.drawLine(
        Offset(cx + r, cy - r), Offset(cx - r, cy + r), paint);
    canvas.drawCircle(
        Offset(cx, cy),
        r + 3,
        Paint()
          ..color = const Color(0xFFE24B4A)
          ..style = PaintingStyle.stroke
          ..strokeWidth = 1);
  }

  void _drawRoute(Canvas canvas, double w, double roomH,
      FloorRoute? route, Color color) {
    if (route == null) return;

    final double cy = roomH + corridorH / 2;
    final double sx = route.startX * w;
    final double ex = route.endX * w;

    final Paint linePaint = Paint()
      ..color = color
      ..strokeWidth = 2.5
      ..style = PaintingStyle.stroke
      ..strokeCap = StrokeCap.round;

    final Path src = Path()
      ..moveTo(sx, cy)
      ..lineTo(ex, cy);
    final Path dashed = Path();
    for (final PathMetric m in src.computeMetrics()) {
      double dist = 0;
      bool draw = true;
      while (dist < m.length) {
        final double len = draw ? 9.0 : 5.0;
        if (draw)
          dashed.addPath(m.extractPath(dist, dist + len), Offset.zero);
        dist += len;
        draw = !draw;
      }
    }
    canvas.drawPath(dashed, linePaint);

    // Directional arrow at end point
    final double dir = ex > sx ? 1.0 : -1.0;
    canvas.drawPath(
      Path()
        ..moveTo(ex + dir * 11, cy)
        ..lineTo(ex, cy - 6)
        ..lineTo(ex, cy + 6)
        ..close(),
      Paint()..color = color,
    );

    // Start circle
    canvas.drawCircle(Offset(sx, cy), 5, Paint()..color = color);
  }

  void _drawMarker(Canvas canvas, Offset center, PWDReport report,
      bool isSelected) {
    final Color color = switch (report.issueType) {
      'ramp' => const Color(0xFF1D9E75),
      'stairs' => const Color(0xFFEF9F27),
      'hazard' => const Color(0xFFE24B4A),
      'exit' => const Color(0xFF378ADD),
      'noise' => const Color(0xFF8B5CF6),
      _ => Colors.grey,
    };
    final String label = switch (report.issueType) {
      'hazard' => '!',
      'noise' => 'N',
      _ => report.issueType[0].toUpperCase(),
    };
    final double r = isSelected ? 12.0 : 9.0;

    if (isSelected) {
      canvas.drawCircle(
          center, r + 5, Paint()..color = color.withOpacity(0.25));
    }
    canvas.drawCircle(center, r, Paint()..color = color);
    canvas.drawCircle(
        center,
        r,
        Paint()
          ..color = Colors.white
          ..style = PaintingStyle.stroke
          ..strokeWidth = 1.5);
    _text(canvas, label, center.dx, center.dy,
        fs: isSelected ? 9 : 7,
        color: Colors.white,
        bold: true,
        maxW: r * 2);
  }

  void _text(Canvas canvas, String text, double cx, double cy,
      {required double fs,
      required Color color,
      bool bold = false,
      required double maxW}) {
    final TextPainter tp = TextPainter(
      text: TextSpan(
          text: text,
          style: TextStyle(
              fontSize: fs,
              color: color,
              fontWeight:
                  bold ? FontWeight.bold : FontWeight.normal,
              height: 1.2)),
      textAlign: TextAlign.center,
      textDirection: TextDirection.ltr,
    )..layout(maxWidth: maxW.clamp(1.0, double.infinity));
    tp.paint(canvas, Offset(cx - tp.width / 2, cy - tp.height / 2));
  }

  @override
  bool shouldRepaint(covariant FloorPlanPainter old) =>
      old.floor != floor ||
      old.mode != mode ||
      old.disability != disability ||
      old.selected?.id != selected?.id ||
      old.reports.length != reports.length;
}
